% Column Width Ratios
\newcommand{\leftcolumnratio}{0.3}
\newcommand{\rightcolumnratio}{0.7}

% Left/Right Column's Right/Left Margin
\setlength{\columnmargin}{5mm}

% Column Widths
\setlength{\leftcolumnwidth}{\leftcolumnratio \paperwidth}
\setlength{\rightcolumnwidth}{\rightcolumnratio \paperwidth}
\setlength{\leftcolumntextwidth}{\dimexpr \leftcolumnwidth - \leftsidemargin - \columnmargin}
\setlength{\rightcolumntextwidth}{\dimexpr \rightcolumnwidth - \rightsidemargin - \columnmargin}

\setlength{\photowidth}{0.9 \leftcolumntextwidth}
