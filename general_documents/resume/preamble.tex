% Main Body Background
\newcommand{\setbackground}[4]{
    \tikz[
        remember picture,
        overlay
    ]
    {
        \node[
            shape          = rectangle,
            fill           = #3,
            anchor         = west,
            minimum width  = #1,
            minimum height = \paperheight,
            inner sep      = 0mm
        ]
        (left column) at (current page.west) {};

        \node[
            shape          = rectangle,
            fill           = #4,
            anchor         = west,
            minimum width  = #2,
            minimum height = \paperheight,
            inner sep      = 0mm
        ]
        (right column) at (left column.east) {};

        \draw[
            color      = black,
            line width = 0.1mm
        ]
        (left column.north east) -- (right column.south west);
    }
}

% Custom Header
\newcommand{\header}[4]{
    \tikz[
        remember picture, 
        overlay
    ]
    {
        \node[
            shape          = rectangle, 
            fill           = #3, 
            anchor         = north, 
            minimum width  = \paperwidth, 
            minimum height = \topsidemargin,
            inner sep      = 0mm
        ] 
        (header) at (current page.north) {};

        \node[
            anchor = center,
        ]
        (name) at (header) {{\Huge \bfseries \color{#4} #1, #2}};

        \draw[
            black,
            line width = 1mm
        ]
        (header.south east) -- (header.south west);
    }
}

                inner sep      = 0mm,

                    \node at (path picture bounding box.center) {
                        \includegraphics[
                            width  = #3,
                            height = #3]
                            {
                                #1
                            }
                    };
                }
            ]

            }
        }

}
