% Custom header
\newcommand{\customheader}[4]{
    \tikz[
        remember picture, 
        overlay
    ]
    {
        \node[
            rectangle, 
            fill           = #3, 
            anchor         = north, 
            minimum width  = \paperwidth, 
            minimum height = 35mm,
            inner sep      = 0mm
        ] 
        (header) at (current page.north) {};

        \node[
            anchor = center,
            font = \Huge \color{#4}
        ]
        (name) at (header) {{#1 \textbf{#2}}};

        \draw[
            black,
            line width = 1mm
        ]
        (header.south east) -- (header.south west);
    }
    \vspace{10mm}
}
