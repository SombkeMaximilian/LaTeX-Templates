% -------------------------------------------------------------------------------------------------
% Header and Footer

\makeatletter

\newcommand*{\makeheader}{
    \begingroup
    \linespread{1.05}
    \selectfont
        \begin{minipage}[t]{0.4 \textwidth}
            \raggedright
            \begin{tabular}{@{}l}
                \@rname \\
                \@raddressone \\
                \@raddresstwo \\
                \@raddressthree \\
                \@raddressfour            
            \end{tabular}
        \end{minipage}%
        \hfill%
        \begin{minipage}[t]{0.4 \textwidth}
            \raggedleft
            \begin{tabular}{l@{}}
                \@sname \\
                \@saddressone \\
                \@saddresstwo \\
                \@saddressthree \\
                \@saddressfour
            \end{tabular}
        \end{minipage}%
        \vspace{2 \baselineskip} \\
        \raggedleft
        \begin{minipage}[t]{0.5 \textwidth}
            \raggedleft
            \@dateinfo
        \end{minipage}%
        \vspace{0.5 \baselineskip} \\
        \raggedright
        \begin{minipage}[t]{0.7 \textwidth}
            \raggedright
            \bfseries Subject: \@subject
        \end{minipage}%
        \vspace{1 \baselineskip}
    \endgroup
}
\newcommand*{\makefooter}{
    \begingroup
    \linespread{1.05}
    \selectfont
    \vfill
        \begin{minipage}[t]{\textwidth}
            \raggedright
            \textbf{Enclosures:} \@enclosures
        \end{minipage}
    \endgroup
}

\makeatother
